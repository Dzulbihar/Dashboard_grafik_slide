CREATE TABLE DASHBOARDGRAFIK.S_TARGET_RKAP_PERBULAN
(
  ID           NUMBER(19)     ,                  
  JENIS_DATA   VARCHAR2(255 BYTE),
  TAHUN        VARCHAR2(255 BYTE),
  BULAN        VARCHAR2(255 BYTE),
  TARGET_RKAP  VARCHAR2(255 BYTE),
  SATUAN       VARCHAR2(255 BYTE),
  TYPE         VARCHAR2(255 BYTE),
  CREATED_AT   TIMESTAMP(6),
  UPDATED_AT   TIMESTAMP(6)
)

CREATE SEQUENCE DASHBOARDGRAFIK.S_TARGET_RKAP_PERBULAN_SEQ START WITH 1;

CREATE TRIGGER DASHBOARDGRAFIK.S_TARGET_RKAP_PERBULAN_ID 
BEFORE INSERT ON DASHBOARDGRAFIK.S_TARGET_RKAP_PERBULAN 
FOR EACH ROW
BEGIN
  SELECT S_TARGET_RKAP_PERBULAN_SEQ.NEXTVAL
  INTO   :new.id
  FROM   dual;
END;
