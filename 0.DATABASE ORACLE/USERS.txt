CREATE TABLE DASHBOARDGRAFIK.S_USERS
(
  ID                 NUMBER(19)                 NOT NULL,
  ROLE               VARCHAR2(255 BYTE)         NOT NULL,
  NAME               VARCHAR2(255 BYTE)         NOT NULL,
  EMAIL              VARCHAR2(255 BYTE)         NOT NULL,
  EMAIL_VERIFIED_AT  TIMESTAMP(6),
  PASSWORD           VARCHAR2(255 BYTE)         NOT NULL,
  REMEMBER_TOKEN     VARCHAR2(100 BYTE),
  CREATED_AT         TIMESTAMP(6),
  UPDATED_AT         TIMESTAMP(6)
)

CREATE SEQUENCE DASHBOARDGRAFIK.S_USERS_SEQ START WITH 1;

CREATE TRIGGER DASHBOARDGRAFIK.S_USERS_ID 
BEFORE INSERT ON DASHBOARDGRAFIK.S_USERS 
FOR EACH ROW
BEGIN
  SELECT S_USERS_SEQ.NEXTVAL
  INTO   :new.id
  FROM   dual;
END;
