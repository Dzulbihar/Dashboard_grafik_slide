CREATE TABLE DASHBOARDGRAFIK.S_SYSCODE
(
  	ID           	NUMBER(19)      ,                 
  	KODE        	VARCHAR2(255 BYTE),
	VALUE_CHAR        VARCHAR2(255 BYTE),
	VALUE_NUMBER      VARCHAR2(255 BYTE),
	KET_CHAR        	VARCHAR2(255 BYTE),
	KET_NUMBER        VARCHAR2(255 BYTE),
	CREATED_AT		TIMESTAMP(6),
  	UPDATED_AT   	TIMESTAMP(6)
)

CREATE SEQUENCE DASHBOARDGRAFIK.S_SYSCODE_SEQ START WITH 1;

CREATE TRIGGER DASHBOARDGRAFIK.S_SYSCODE_ID 
BEFORE INSERT ON DASHBOARDGRAFIK.S_SYSCODE 
FOR EACH ROW
BEGIN
  SELECT S_SYSCODE_SEQ.NEXTVAL
  INTO   :new.id
  FROM   dual;
END;
