CREATE TABLE DASHBOARDGRAFIK.S_PROD_PEND_PERBULAN
(
  ID                NUMBER,
  JENIS_DATA        CHAR(8 BYTE),
  LOKASI            VARCHAR2(13 BYTE),
  TAHUN             VARCHAR2(4 BYTE),
  BULAN             VARCHAR2(2 BYTE),
  SHIPCALL          NUMBER,
  GT                NUMBER,
  JML_BOX           NUMBER,
  JML_TEUS          NUMBER,
  TOTAL_PENDAPATAN  NUMBER,
  TAHUN_DEPARTURE   NUMBER,
  BULAN_DEPARTURE   NUMBER
)

CREATE SEQUENCE DASHBOARDGRAFIK.S_PROD_PEND_PERBULAN_SEQ START WITH 1;

CREATE TRIGGER DASHBOARDGRAFIK.S_PROD_PEND_PERBULAN_ID 
BEFORE INSERT ON DASHBOARDGRAFIK.S_PROD_PEND_PERBULAN
FOR EACH ROW
BEGIN
  SELECT S_PROD_PEND_PERBULAN_SEQ.NEXTVAL
  INTO   :new.id
  FROM   dual;
END;
